                'results_returned': facetedProducts.totalCount,
                'category_page_name': pageContext.title,
                'category_page_id': pageContext.categoryId
            });
            if(pageContext.crawlerInfo.canonicalUrl && pageContext.crawlerInfo.canonicalUrl.indexOf("brands") != -1){
                dataLayerObj.ecommerce.search_brand = pageContext.title;
            }
        } else if(pageContext.pageType == "search") {
            dataLayerObj.ecommerce = Object.assign(dataLayerObj.ecommerce, {
                'search_keyword': pageContext.search.query,
                'results_returned': facetedProducts ? facetedProducts.totalCount : 0,
                'search_category_id': pageContext.search.categoryId || ""
            });
        } else if(customer){
            for(var contactId in customer.contacts){
                if(customer.contacts[contactId].types.length && customer.contacts[contactId].types[0].isPrimary){
                    dataLayerObj.ecommerce = Object.assign(dataLayerObj.ecommerce, {
                        'billing_city': customer.contacts[contactId].address.cityOrTown,
                        'billing_country': customer.contacts[contactId].address.countryCode,
                        'billing_state': customer.contacts[contactId].address.stateOrProvince,
                        'billing_zip': customer.contacts[contactId].address.postalOrZipCode
                    });
                }
            }
        } else if(pageContext.pageType == "product"){
            var brand = "none",
                    rating = 0;
            for(var propertyIndex = 0; propertyIndex < product.properties.length; propertyIndex++){
                if(product.properties[propertyIndex].attributeFQN == "tenant~A00413"){
                    brand = product.properties[propertyIndex].values[0].stringValue;
                }
            }
            for(var propertyIndex = 0; propertyIndex < product.properties.length; propertyIndex++){
                if(product.properties[propertyIndex].attributeFQN == "tenant~rating"){
                    rating = product.properties[propertyIndex].values[0].value;
                }
            }

            dataLayerObj.ecommerce = Object.assign(dataLayerObj.ecommerce, {
                "detail": {
                    "products": [{
                        "id": product.productCode,
                        "name": product.content.productName,
                        "category": product.categories.length ? product.categories[0].categoryId : 0,
                        "price": product.price.salePrice ? product.price.salePrice : product.price.price,
                        "url": product.url,
                        "dimension1": rating,
                        "brand": brand
                    }]
                }
            });
        } 

        if(!window.dataLayer) window.dataLayer = [];
        if(pageContext.pageType !== 'cart') window.dataLayer.push(dataLayerObj);
    }

} catch (error) {
    console.log(error);
